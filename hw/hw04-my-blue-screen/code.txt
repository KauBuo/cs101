image = new SimpleImage("monkey.jpg")
back = new SimpleImage("背景.jpg")
for (p:image){
avg = (p.getRed()+p.getBlue()+p.getGreen())/3
if (p.getBlue() > avg*0.90){
x = p.getX()
y = p.getY()

j = back.getPixel(x+420,y+140)

p.setRed(j.getRed())
p.setGreen(j.getGreen())
p.setBlue(j.getBlue())
}

}

print(image)
